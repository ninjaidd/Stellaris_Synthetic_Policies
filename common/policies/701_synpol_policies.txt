# Cybernetics Policy
syn_pol_cybernetic_policy = {

	potential = {
		NOT = { has_ethic = ethic_gestalt_consciousness }
        has_ascension_perk = ap_synpol_the_flesh_is_weak
        NOT = { has_country_flag = flesh_weakened }
	}

	option = {
		name = "cybernetics_optional"

		policy_flags = {
			cybernetics_optional
		}
    }

    option = {
		name = "cybernetics_discretionary"

		valid = {
			NOR = {
                has_ethic = "ethic_egalitarian"
                has_ethic = "ethic_fanatic_egalitarian"
            }
		}

		policy_flags = {
			cybernetics_discretionary
        }

		ai_weight = { weight = 0 }
    }

    option = {
		name = "cybernetics_compulsory"

		valid = {
			NOR = {
                has_ethic = "ethic_egalitarian"
                has_ethic = "ethic_fanatic_egalitarian"
            }
            NOT = { has_policy_flag = synthetics_compulsory}
		}

		policy_flags = {
			cybernetics_compulsory
        }

        ai_weight = {
            modifier = {
                factor = 2
                OR = {
                    has_ethic = "ethic_materialist"
                    has_ethic = "ethic_fanatic_materialist"
                    has_ethic = "ethic_authoritarian"
                    has_ethic = "ethic_fanatic_authoritarian"
                }
			}
            modifier = {
                factor = 3
                has_ethic = "ethic_materialist"
                has_ethic = "ethic_authoritarian"                
			}
            modifier = {
                factor = 5
                has_ethic = "ethic_fanatic_materialist"
                has_ethic = "ethic_fanatic_authoritarian"              
			}
        }
    }
}
